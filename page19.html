SyntaxHighlighter動作テスト

FizzBuzzで試してみます。
#!/usr/bin/env python
# -*- coding:utf-8 -*-

def fizzbuzz(n):
    if n % 15 == 0:
        ans =  'fizzbuzz'
    elif n % 3 == 0:
        ans =  'fizz'
    elif n % 5 == 0:
        ans =  'buzz'
    else:
        ans =  n
    return ansdef main():

for i in range(1, 101):
    print fizzbuzz(i)

if __name__ == '__main__':
    main()
